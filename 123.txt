    // Scroll reveal

    revealEls.forEach(el => observer.observe(el));

    // ---- Auth Modal ----
    let currentRole = 'creator';
    let currentTab  = 'login';

    const config = {
      creator: {
        badge: 'Content Creator',
        accentClass: 'creator',

        loginTitle: 'Welcome back',
        loginSubtitle: 'Log in to browse briefs and manage your active projects.',

        signupSubtitle: 'Join editors already earning steady income from creators who need your craft.',
        submitClass: 'editor-btn',

    function openAuth(role, e) {
      if (e) e.preventDefault();
      currentRole = role;
      currentTab  = 'login';

      // FIX 2: fully reset success state before opening
      const s = document.getElementById('modalSuccess');
      s.classList.remove('visible');
      s.style.display = 'none';

      applyRole(role);
      switchTab('login');

      document.getElementById('authTabs').style.display    = 'flex';
      document.getElementById('formLogin').style.display   = 'block';
      document.getElementById('formSignup').style.display  = 'none';
      document.getElementById('authModal').classList.add('active');
      document.body.style.overflow = 'hidden';
    }

    function applyRole(role) {
      const c = config[role];
      const isEditor = role === 'editor';

      // Accent bar
      document.getElementById('modalAccent').className = 'modal-accent ' + c.accentClass;

      // FIX 1: set badge className only once on the correct element
      const badge = document.getElementById('modalBadge');
      badge.className = 'modal-role-badge ' + c.accentClass;
      document.getElementById('modalBadgeText').textContent = c.badge;

      // Titles
      updateTitles(role, currentTab);

      // Button styles
      document.getElementById('loginSubmitBtn').className  = 'auth-submit ' + c.submitClass;
      document.getElementById('signupSubmitBtn').className = 'auth-submit ' + c.submitClass;
      document.getElementById('successBtn').className      = 'auth-submit ' + c.submitClass;

      // Editor-specific field
      document.getElementById('editorField').style.display = isEditor ? 'flex' : 'none';

      // Focus ring colour
      document.getElementById('signupForm').classList.toggle('editor-mode', isEditor);
    }

    function updateTitles(role, tab) {
      const c = config[role];
      document.getElementById('modalTitle').textContent    = tab === 'login' ? c.loginTitle    : c.signupTitle;
      document.getElementById('modalSubtitle').textContent = tab === 'login' ? c.loginSubtitle : c.signupSubtitle;
    }

    function switchTab(tab) {
      currentTab = tab;
      document.getElementById('tabLogin').classList.toggle('active',  tab === 'login');
      document.getElementById('tabSignup').classList.toggle('active', tab === 'signup');
      document.getElementById('formLogin').style.display  = tab === 'login'  ? 'block' : 'none';
      document.getElementById('formSignup').style.display = tab === 'signup' ? 'block' : 'none';
      updateTitles(currentRole, tab);
    }

    function closeAuth() {
      document.getElementById('authModal').classList.remove('active');
      document.body.style.overflow = '';
    }

    function handleOverlayClick(e) {
      if (e.target === document.getElementById('authModal')) closeAuth();
    }

    function handleSubmit() {
      const c = config[currentRole];
      document.getElementById('authTabs').style.display    = 'none';
      document.getElementById('formLogin').style.display   = 'none';
      document.getElementById('formSignup').style.display  = 'none';
      document.getElementById('modalTitle').textContent    = '';
      document.getElementById('modalSubtitle').textContent = '';

      const success = document.getElementById('modalSuccess');
      document.getElementById('successIcon').textContent  = c.successIcon;
      document.getElementById('successIcon').className    = 'success-icon ' + (currentRole === 'creator' ? 'creator-success-icon' : 'editor-success-icon');
      document.getElementById('successTitle').textContent = c.successTitle;
      document.getElementById('successSub').textContent   = c.successSub;
      success.style.display = 'flex';
      success.classList.add('visible');
    }

    function updatePwdStrength(val) {
      const score = (val.length >= 8 ? 1 : 0)
                  + (/[A-Z]/.test(val) ? 1 : 0)
                  + (/[0-9]/.test(val) ? 1 : 0)
                  + (/[^A-Za-z0-9]/.test(val) ? 1 : 0);
      const colors = ['#E24B4A','#E8A020','#C8FF00','#5DCAA5'];
      ['bar1','bar2','bar3','bar4'].forEach((id, i) => {
        document.getElementById(id).style.background = i < score ? colors[score - 1] : 'rgba(255,255,255,0.08)';
      });
    }

  <script>
    // ─── Dashboard JS ───

    function showEditorDashboard() {
      // Hide homepage, show dashboard
      document.querySelector('nav').style.display = 'none';
      document.querySelector('.hero')              && (document.querySelector('.hero').style.display = 'none');
      document.querySelectorAll('.stats-bar, .section, .profiles-section, .testimonials, .final-cta, footer')
        .forEach(el => el.style.display = 'none');
      document.getElementById('authModal').classList.remove('active');
      document.body.style.overflow = '';
      document.body.style.display  = 'block';

      const db = document.getElementById('editorDashboard');
      db.classList.add('visible');

      dbRenderBars(dbMonthData);

      // Timeline hover scrub
      db.querySelectorAll('.db-timeline-strip').forEach(strip => {
        strip.addEventListener('mousemove', e => {
          const rect = strip.getBoundingClientRect();
          const pct  = ((e.clientX - rect.left) / rect.width * 100).toFixed(1);
          const head = strip.querySelector('.db-tl-head');
          if (head) head.style.left = pct + '%';
        });
      });
    }

    function exitDashboard() {
      document.getElementById('editorDashboard').classList.remove('visible');
      // Restore homepage
      document.querySelector('nav').style.display = '';
      document.querySelectorAll('.stats-bar, .section, .profiles-section, .testimonials, .final-cta, footer')
        .forEach(el => el.style.display = '');
      document.querySelector('.hero') && (document.querySelector('.hero').style.display = '');
      document.body.style.display = '';
    }

    function dbSetActive(el) {
      document.querySelectorAll('#editorDashboard .db-nav-item').forEach(n => n.classList.remove('active'));
      el.classList.add('active');
    }

    function dbToggleAvail() {
      const t = document.getElementById('dbAvailToggle');
      t.classList.toggle('open');
      document.getElementById('dbAvailLabel').textContent = t.classList.contains('open') ? 'OPEN' : 'AWAY';
    }

    function dbPulse(el) {
      el.style.transform = 'scale(0.95)';
      setTimeout(() => el.style.transform = '', 120);
    }

    const dbMonthData = [
      {label:'1',  v:52},{label:'5',  v:38},{label:'9',  v:65},
      {label:'12', v:44},{label:'15', v:80},{label:'18', v:58},
      {label:'20', v:72},{label:'22', v:90},{label:'24', v:62},
      {label:'26', v:55},{label:'28', v:78},{label:'30', v:100},
    ];
    function dbRenderBars(data) {
      const chart = document.getElementById('dbBarChart');
      if (!chart) return;
      chart.innerHTML = '';
      const maxV = Math.max(...data.map(d => d.v));
      data.forEach((d, i) => {
        const col = document.createElement('div'); col.className = 'db-bar-col';
        const bar = document.createElement('div'); bar.className = 'db-bar' + (i === data.length - 1 ? ' active-bar' : '');
        bar.style.height = (d.v / maxV * 100) + '%';
        const lbl = document.createElement('div'); lbl.className = 'db-bar-label'; lbl.textContent = d.label;
        col.appendChild(bar); col.appendChild(lbl); chart.appendChild(col);
      });
    }
    function dbSetChartTab(el) {
      document.querySelectorAll('#editorDashboard .db-chart-tab').forEach(t => t.classList.remove('active'));
      el.classList.add('active');
      const shuffled = dbMonthData.map(d => ({...d, v: Math.floor(Math.random()*60)+40}));
      dbRenderBars(shuffled);
    }

    // Override handleSubmit to route editors to dashboard
    const _origHandleSubmit = handleSubmit;
    handleSubmit = function() {
      if (currentRole === 'editor') {
        closeAuth();
        showEditorDashboard();
      } else {
        _origHandleSubmit();
      }
    };
  </script>
